## input checked
index.html
```
  <input type="checkbox" id="inp">
  <label for="inp">点击</label>
```
用dom|jQuery判断选中
```
$(function() {
  let $inp = $("#inp");
  let inp = $inp[0];
  $inp.click(function() {
    if(inp.checked) { // dom
    if($inp.is(":checked")) { // jQuery
      alert("ok");
    }
  })
})
