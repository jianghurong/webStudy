## 父组件向子组件传值
父组件
```html
<template>
    <div>
        <Header :activeIndex="index"></Header>
    </div>
</template>
```
```ts
<script>
    import Header from "../Header";
    export default {
        components: {
            Header
        },
        data () {
            return {
                index: "1"
            }
        },
    }
</script>
```
子组件
```ts
<script>
    export default {
        props: [
            "activeIndex"
        ]
    }
</script>
```

## 子组件向父组件传值
子组件
```html
<template>
    <div>
        <p @click="postIndex(2)">点击</p>
    </div>
</template>
```
```js
<script>
    export default {
        methods: {
            postIndex: function(index) {
                this.$emit("getIndex", index);
            }
        }
    }
</script>
```
父组件
```html
<template>
    <div>
        <Header @getIndex="getIndex"></Header>
    </div>
</template>
```
```ts
<script>
    export default {
        methods: {
            getIndex: function(index) {
                alert(index);
            }
        }
    }
</script>
```
